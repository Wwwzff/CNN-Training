      class LossHistory(Callback):
          def on_train_begin(self,logs={}):
              self.losses = []
              self.val_losses = []
              
          def on_epoch_end(self,batch,logs={}):
              self.losses.append(logs.get('loss'))
              self.val_losses.append(logs.get('val_loss'))
